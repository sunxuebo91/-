## 二、快速开始（5步上手）

### 第一步：创建数据库集合（5分钟）

1. 打开微信开发者工具
2. 点击"云开发"按钮
3. 进入"数据库"标签
4. 创建以下集合：

#### 创建 babyDiaries 集合
```bash
集合名称：babyDiaries
权限设置：仅创建者可读写
```

#### 创建 serviceContracts 集合
```bash
集合名称：serviceContracts
权限设置：仅创建者可读写
```

5. 创建索引（在云开发控制台 - 数据库 - 索引管理）：

**babyDiaries 索引**：
```javascript
// 索引1：按合同查询

// 索引2：月嫂查询某日日记

```

```javascript
// 索引1：查询月嫂的活跃合同
```

```javascript
cloud.init({
  env: cloud.DYNAMIC_CURRENT_ENV,

const _ = db.command;

// 权限校验函数
async function isStaff(openid) {
  const staffDoc = await db.collection("staff")
    .where({ openid })
    .limit(1)
    .get();
  return staffDoc.data && staffDoc.data.length > 0;
}

// 主函数
exports.main = async (event, context) => {
  const { action, data } = event;
  const { OPENID } = cloud.getWXContext();

  try {
    switch (action) {
      case "createDiary":
        return await createDiary(OPENID, data);
      case "updateDiary":
        return await updateDiary(OPENID, data);
      case "getDiary":
        return await getDiary(OPENID, data);
      case "listDiaries":
        return await listDiaries(OPENID, data);
      case "deleteDiary":
        return await deleteDiary(OPENID, data);
      case "createContract":
        return await createContract(OPENID, data);
      case "listContracts":
        return await listContracts(OPENID, data);
      default:
        return { success: false, message: "未知操作" };
    }
  } catch (error) {
    console.error("云函数执行错误:", error);
    return { success: false, message: error.message };
  }
};
```

## 四、常见问题

### Q1: 云函数调用失败？
**A**: 检查云函数是否已部署，权限是否正确配置。

### Q2: 数据库查询为空？
**A**: 检查是否已创建测试数据，索引是否已创建。

### Q3: 首页入口不显示？
**A**: 检查用户角色是否为 staff，wx:if 条件是否正确。

### Q4: 如何调试云函数？
**A**: 在云开发控制台 - 云函数 - 日志中查看执行日志。
